#   flake8 app/                               # Linting

#   safety check                              # Vulnerabilidades
#
# Formatação:
#   black app/                                # Formatar código
#   isort app/                                # Organizar imports
#   black --check app/                        # Verificar sem alterar
#
# Pre-commit:
#   pre-commit install                        # Instalar hooks
#   pre-commit run --all-files                # Rodar em todos os arquivos
#
# Profiling:
#   python -m cProfile -o output.prof app.py  # Profile
#   snakeviz output.prof                      # Visualizar
#   py-spy record -o profile.svg -- python app.py  # Sample profiling
